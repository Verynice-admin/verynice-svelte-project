export async function seedHistory() {
  const pageRef = db.collection('pages').doc('historyPage');
  const sectionsRef = pageRef.collection('sections');

  let updated = 0;

  for (const section of historySectionsToSeed) {
    const { id, ...data } = section;
    const docRef = sectionsRef.doc(id);
    const existing = await docRef.get();
    const payload = {
      ...data,
      order: data.order ?? 0,
      updatedAt: new Date()
    };
    if (!existing.exists || !existing.data()?.createdAt) {
      payload.createdAt = new Date();
    }
    await docRef.set(payload, { merge: true });
    updated++;
  }

  await pageRef.set(
    {
      lastHistorySeedAt: new Date()
    },
    { merge: true }
  );

  console.log(`✅ Seeded ${updated} history section${updated === 1 ? '' : 's'}.`);
}
